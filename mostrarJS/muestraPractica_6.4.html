4. Crea una aplicación web que muestre el texto: velocidad 0. Si pulsamos la tecla de la flecha hacia arriba del teclado, la velocidad se incremente en 1. Si es la flecha hacia abajo, la velocidad baja en 1. La velocidad no puede sobrepasar los 120 ni descender de cero:


                          document.addEventListener("DOMContentLoaded", function () {
                            var velocidadElement = document.getElementById('velocidad');
                            var velocidadSpan = velocidadElement.querySelector('span');
                            var velocidad = 0;

                          document.addEventListener('keydown', function (event) {
                            if (event.key === 'ArrowUp') {
                              velocidad = Math.min(velocidad + 1, 120);
                            } else if (event.key === 'ArrowDown') {
                              velocidad = Math.max(velocidad - 1, 0);
                            }

                            velocidadSpan.textContent = velocidad;
                          });

                          });
