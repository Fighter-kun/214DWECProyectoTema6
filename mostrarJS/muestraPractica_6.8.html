                          document.addEventListener("DOMContentLoaded", function () {
                            const boton1 = document.querySelector(".opciones button:first-of-type");
                            const boton2 = document.querySelector("button:last-of-type");
                            const tabla1 = document.querySelector(".OSI");
                            const tabla2 = document.querySelector(".TCP");
                            const primerCheckbox = document.querySelector('.opciones input[type="checkbox"][name="opcion"][value="OSI"]');
                            const segundoCheckbox = document.querySelector('.opciones input[type="checkbox"][name="opcion"][value="TCP"]');
                            const overlay = document.querySelector('.overlay');

                            primerCheckbox.addEventListener('change', function () {
                                if (this.checked) {
                                    // Si el checkbox está desmarcado, ocultar los elementos OSI
                                    tabla1.style.display = 'block';
                                } else {

                                    // Si el checkbox está marcado, mostrar los elementos OSI
                                    tabla1.style.display = 'none';
                                }
                            });
                            segundoCheckbox.addEventListener('change', function () {
                                if (this.checked) {
                                    // Si el checkbox está desmarcado, ocultar los elementos TCP
                                    tabla2.style.display = 'block';
                                } else {
                                    // Si el checkbox está marcado, mostrar los elementos TCP
                                    tabla2.style.display = 'none';
                                }
                            });

                            boton1.addEventListener('click', function () {
                                tabla1.style.display = 'none';
                                tabla2.style.display = 'none';
                                primerCheckbox.checked = false;
                                segundoCheckbox.checked = false;
                            });

                            boton2.addEventListener('click', function () {
                                overlay.classList.add('active');
                                overlay.addEventListener('click', function () {
                                    overlay.classList.remove('active');
                                });
                            });
                        });
