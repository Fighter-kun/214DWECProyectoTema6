3. Crear una aplicación web que muestre una capa centrada que ocupe el 50% del ancho y el alto de la ventana...:


                          document.addEventListener("DOMContentLoaded", function () {
                            // Obtén la referencia al elemento que seguirá al cursor
                            var elementoCapa = document.getElementById('capa');
                            // Agrega un event listener para el punto B
                            elementoCapa.addEventListener('mouseover', function (event) {
                              elementoCapa.style.backgroundColor = 'green';
                            });
                            // Para revertir el punto B
                            elementoCapa.addEventListener('mouseout', function (event) {
                              elementoCapa.style.backgroundColor = 'white';
                            });
                            // Agrega un event listener para el punto C y D
                            elementoCapa.addEventListener('mousedown', function (event) {
                              if (event.button == '0') {
                                elementoCapa.style.backgroundColor = 'red';
                              } else {
                                if (event.button == '2') {
                                elementoCapa.style.backgroundColor = 'blue';
                              }
                              }
                            });
                            // Para revertir el punto C
                            elementoCapa.addEventListener('mouseup', function (event) {
                              elementoCapa.style.backgroundColor = 'green';
                            });
                            // Quitar cuadro al presionar el boton 2 del ratón
                            elementoCapa.addEventListener('contextmenu', function (event) {
                              event.preventDefault();
                            });
                          });
